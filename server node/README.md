function legalTz(tz)
{
    let total = 0;
    let i=0;
    for (i = 0; i < 9; i++) {
        let x = (((i % 2) + 1) * tz.charAt(i));
        total +=Math.floor(x/10) + x%10;
    }
    return (total % 10 == 0);
}
